# Go + Node.js development environment
FROM codercom/code-server:latest

# Metadata
LABEL description="MCS Go + Node.js development environment"

# Switch to root for installations
USER root

# Install Node.js and npm
RUN curl -fsSL https://deb.nodesource.com/setup_20.x | bash - && \
    apt-get install -y nodejs && \
    rm -rf /var/lib/apt/lists/*

# Install Go
ENV GO_VERSION=1.21.5
RUN wget -O go.tar.gz "https://golang.org/dl/go${GO_VERSION}.linux-amd64.tar.gz" && \
    tar -C /usr/local -xzf go.tar.gz && \
    rm go.tar.gz

# Set Go environment variables
ENV PATH="/usr/local/go/bin:${PATH}"
ENV GOPATH="/home/coder/go"
ENV PATH="${GOPATH}/bin:${PATH}"

# Install Go tools
RUN go install golang.org/x/tools/gopls@latest && \
    go install github.com/go-delve/delve/cmd/dlv@latest

# Install common npm packages
RUN npm install -g \
    yarn \
    typescript \
    nodemon

# Switch back to coder user
USER coder

# Set up workspaces
RUN mkdir -p ~/go/{bin,src,pkg} && \
    npm config set prefix "$HOME/.npm-global" && \
    mkdir -p ~/.local/bin ~/.local/share ~/.npm-global/bin

# Verify installations
RUN go version && node --version && npm --version
